255void updateCurrentSelect(uint8_t out_b) {
256
257 switch (out_b) {
258 case KBD_BREAKCODE_ARROW_DOWN: // DOWN
259 if (db->currentSelect == 2)
260 db->currentSelect = 0;
261 else
262 db->currentSelect++;
263 break;
264 case KBD_BREAKCODE_ARROW_UP: // UP
265 if (db->currentSelect == 0)
266 db->currentSelect = 2;
267 else
268 db->currentSelect--;
269 break;
270 case KBD_BREAKCODE_ENTER: // ENTER
271 switch (db->currentSelect) {
272 case 0:
273 db->currentState = GamePlaying;
274 break;
275 case 1:
276 db->currentState = Menu_Rules;
277 break;
278 case 2:
279 db->currentState = Exit;
280 break;
281 default: break;
282 }
283 default: break;
284 }
285}
